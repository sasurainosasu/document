version: '3.8'

services:
  #--------------------------------
  # リバースプロキシ
  #--------------------------------
  nginx-proxy:
    image: jwilder/nginx-proxy
    container_name: nginx-proxy
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - /var/run/docker.sock:/tmp/docker.sock:ro
      - certs:/etc/nginx/certs:ro
      - vhost.d:/etc/nginx/vhost.d
      - html:/usr/share/nginx/html
    networks:
      - app-network
    restart: always

  #--------------------------------
  # SSL証明書自動取得コンテナ
  #--------------------------------
  acme-companion:
    image: nginxproxy/acme-companion
    container_name: nginx-proxy-acme
    volumes:
      - certs:/etc/nginx/certs
      - vhost.d:/etc/nginx/vhost.d
      - html:/usr/share/nginx/html
      - /var/run/docker.sock:/var/run/docker.sock:ro
    environment:
      - DEFAULT_EMAIL=your-email@example.com # 👈 あなたのメールアドレスに置き換えてください
    networks:
      - app-network
    depends_on:
      - nginx-proxy
    restart: always

  #--------------------------------
  # PHP-FPM
  #--------------------------------
  php:
    container_name: prod-php
    build:
      context: .
      dockerfile: ./docker/php/Dockerfile
    volumes:
      - ./app:/var/www/html
    environment:
      # MailHogの設定をMailserverに変更
      - SMTP_HOST=mailserver
      - SMTP_PORT=587
      - SMTP_USER=your-email@example.com # 👈 Mailserverに設定したメールアカウント
      - SMTP_PASS=your-password # 👈 Mailserverに設定したパスワード
      - VIRTUAL_HOST=api.example.com          # 👈 APIのサブドメインに置き換えてください
      - LETSENCRYPT_HOST=api.example.com      # 👈 APIのサブドメインに置き換えてください
      - LETSENCRYPT_EMAIL=your-email@example.com # 👈 あなたのメールアドレスに置き換えてください
    env_file:
      - .env
    networks:
      - app-network
    depends_on:
      - mailserver # mailserverへの依存関係を追加
    restart: always

  #--------------------------------
  # Next.js (フロントエンド)
  #--------------------------------
  frontend:
    container_name: prod-frontend
    build:
      context: .
      dockerfile: ./docker/frontend/prod.Dockerfile
    volumes:
      - ./app/frontend/.next:/app/.next
      - ./app/frontend/public:/app/public
      - /app/node_modules
    expose:
      - "3000"
    environment:
      - CHOKIDAR_USEPOLLING=true
      - WATCHPACK_POLLING=true
      - VIRTUAL_HOST=app.example.com          # 👈 アプリケーションのサブドメインに置き換えてください
      - LETSENCRYPT_HOST=app.example.com      # 👈 アプリケーションのサブドメインに置き換えてください
      - LETSENCRYPT_EMAIL=your-email@example.com # 👈 あなたのメールアドレスに置き換えてください
    networks:
      - app-network
    tty: true
    restart: always
   
  #--------------------------------
  # MySQL (データベース)
  #--------------------------------
  mysql:
    container_name: prod-mysql
    image: mysql:8.0
    ports:
      - "3306:3306"
    environment:
      MYSQL_DATABASE: ${DB_NAME}
      MYSQL_USER: ${DB_USER}
      MYSQL_PASSWORD: ${DB_PASSWORD}
      MYSQL_ROOT_PASSWORD: rootpassword
      TZ: Asia/Tokyo
    volumes:
      - db-data:/var/lib/mysql
      - ./docker/mysql/my.cnf:/etc/mysql/conf.d/my.cnf
      - ./docker/mysql/sql:/docker-entrypoint-initdb.d
    networks:
      - app-network

  #--------------------------------
  # phpMyAdmin (DB管理ツール)
  #--------------------------------
  db-admin:
    container_name: prod-db-admin
    image: phpmyadmin/phpmyadmin
    ports:
      - "8080:80"
    environment:
      PMA_HOST: mysql
      PMA_PORT: 3306
    depends_on:
      - mysql
    networks:
      - app-network
       
  #--------------------------------
  # Mailserver
  #--------------------------------
  mailserver:
    image: docker.io/mailserver/docker-mailserver:latest
    container_name: mailserver
    hostname: mail.yourdomain.com # 👈 あなたのドメインに置き換えてください
    ports:
      - "25:25"   # SMTP
      - "143:143" # IMAP
      - "587:587" # SMTPS
      - "993:993" # IMAPS
    volumes:
      - ./docker-data/mailserver/mail-data:/var/mail
      - ./docker-data/mailserver/mail-state:/var/mail-state
      - ./docker-data/mailserver/mail-logs:/var/log/mail
      - ./docker-data/mailserver/config/:/tmp/docker-mailserver/
    environment:
      - DMS_DEBUG=1
      - POSTMASTER_ADDRESS=postmaster@yourdomain.com # 👈 あなたのドメインに置き換えてください
      - SMTPS_ONLY=1
      - SSL_TYPE=self-signed # 👈 本番環境では letsencrypt を推奨
    cap_add:
      - NET_ADMIN
    restart: always

# 名前付きボリュームの定義
volumes:
  db-data:
  certs:
  vhost.d:
  html:
  docker-data/mailserver/mail-data:
  docker-data/mailserver/mail-state:
  docker-data/mailserver/mail-logs:
  docker-data/mailserver/config:

# サービスが接続するネットワークの定義
networks:
  app-network:
    driver: bridge
